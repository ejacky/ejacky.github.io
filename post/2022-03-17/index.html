这是用来测试用的

package test

import (
	"fmt"
	"testing"
	"time"
)

func TestBreak(t *testing.T) {
	tick := time.Tick(time.Second)
	for {
		select {
		case t := <-tick:
			fmt.Println(t)    
			break
		}
	}
	fmt.Println("end")
}
